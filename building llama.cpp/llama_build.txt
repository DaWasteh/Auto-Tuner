cd L:\LAB\ai-local
git clone https://github.com/KhronosGroup/SPIRV-Headers.git
cmake -S .\SPIRV-Headers -B .\SPIRV-Headers\build `
  -G "Visual Studio 18 2026" `
  -A x64 `
  -DCMAKE_INSTALL_PREFIX="L:/LAB/ai-local/SPIRV-Headers/install"
cmake --build .\SPIRV-Headers\build --config Release
cmake --install .\SPIRV-Headers\build --config Release

git clone https://github.com/ggml-org/llama.cpp.git _tmp_llama
Push-Location .\_tmp_llama
$desc = git describe --tags --always 2>$null
$ver  = [regex]::Match($desc, 'b\d+').Value
if (-not $ver) { $ver = "bUNKNOWN" }
Pop-Location
$dir = "${ver}_llama.cpp"
if (Test-Path .\$dir) { Remove-Item .\$dir -Recurse -Force }
Rename-Item .\_tmp_llama $dir
Write-Host "==> Build-Verzeichnis: $dir"

Push-Location .\$dir\tools\ui
npm ci
npm run build
Pop-Location

cmake -S .\$dir -B .\$dir\build `
  -G "Visual Studio 18 2026" `
  -A x64 `
  -DGGML_VULKAN=ON `
  -DGGML_NATIVE=OFF `
  -DGGML_AVX2=ON `
  -DGGML_FMA=ON `
  -DGGML_F16C=ON `
  -DBUILD_SHARED_LIBS=OFF `
  -DCMAKE_POLICY_DEFAULT_CMP0194=OLD `
  -DCMAKE_ASM_COMPILER=cl `
  -DLLAMA_BUILD_SERVER=ON `
  -DLLAMA_BUILD_UI=ON `
  -DLLAMA_USE_PREBUILT_UI=OFF `
  -DLLAMA_CURL=OFF `
  -DGGML_CCACHE=OFF `
  -DGGML_VULKAN_CHECK_RESULTS=OFF `
  -DCMAKE_PREFIX_PATH="L:/LAB/ai-local/SPIRV-Headers/install"
cmake --build .\$dir\build --config Release --parallel 24
